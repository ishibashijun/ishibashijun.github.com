def header(self):
    ret = pyNodeList()
    ret.extend(self._nodes.tag('h1'))
    ret.extend(self._nodes.tag('h2'))
    ret.extend(self._nodes.tag('h3'))
    ret.extend(self._nodes.tag('h4'))
    ret.extend(self._nodes.tag('h5'))
    ret.extend(self._nodes.tag('h6'))
    return ret
