__eq__

returns true if it equals

Example


parser = Parser()
parser.open("http://example.com")
div_main = parser.tag('div').id('main')
id_main = parser.id('main')
print(str(div_main == id_main))
parser.close()

>> True

Source


def __eq__(self, node):
    if self._name == node._name and \
       self._attr == node._attr and \
       self._html == node._html and \
       self._parent == node._parent and \
       self._children == node._children:
        return True
    return False
