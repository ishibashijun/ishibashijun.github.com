select_next_sibling

returns the sibling that is from the next element.

Example


Q_open("http://example.com")
second = Q('ul:first > li:nth-child(2)')
from_third_li = second.select_next_sibling()

def select_next_sibling(self):
    ret = pyNodeList()
    n = self.next_all()
    if isinstance(n, pyNode):
        ret.append(n)
    elif isinstance(n, pyNodeList) and len(n) != 0:
        for elem in n:
            ret.append(elem)
    if len(ret) == 1: return ret[0]
    else: return ret
