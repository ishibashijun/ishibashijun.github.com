next_all

returns elements after this elements that in same DOM hierarchy

Example


parser = Parser()
parser.open("http://example.com")
li = Parser.tag('ul').first().child('li').eq(3).next_all()
parser.close()

Source


def next_all(self):
    ret = pyNodeList()
    if self.has_parent():
        brothers = self._parent.children()
        pass_me = False
        for i in range(len(brothers)):
            if brothers[i].name() is not 'comment':
                bro = brothers[i]
                if pass_me: ret.append(bro)
                if bro is self: pass_me = True
    return ret
