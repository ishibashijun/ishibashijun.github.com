child

returns specified child elements or children

Example


parser = Parser()
parser.open("http://example.com")
li = parser.tag('ul').first().child('li')
parser.close()

Arguments

  • Name: child_tag, Type: str
    • child element tag name

Source


def child(self, child_tag=None):
    if childTag is None: return self._children
    ret = pyNodeList()
    for node in self._children:
        if node.name() == childTag:
            ret.append(node)
    return ret
