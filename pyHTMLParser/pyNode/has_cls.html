has_cls

returns True if element has specified class attribute

Example


parser = Parser()
parser.open("http://example.com")
has_link_class = parser.tag('a').first().has_cls("link")
parser.close()

Arguments

  • Name: cls, Type: str

Source


def has_cls(self, cls):
    if 'class' in self._attr:
        cls_name = re.compile(cls)
        if cls_name.search(self._attr['class']) is not None:
            return True
    return False
