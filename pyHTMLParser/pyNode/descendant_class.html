descendant_class

returns descendants of a element matching a class name

Example


parser = Parser()
parser.open("http://example.com")
ul = parser.tag('ul')
links = ul.descendant_class('link')
parser.close()

Arguments

  • Name: cls, Type: str
    • class name

Source


def descendant_class(self, cls):
    ret = pyNodeList()
    cls_name = re.compile('(' + cls + ')')
    if self.has_child() == True:
        children = self.children()
        for ch in children:
            if ch.attr('class') is not None and cls_name.search(ch.attr('class')) is not None:
                ret.append(ch)
            descendant = ch.descendant_class(cls)
            if len(descendant) != 0:
                ret.extend(descendant)
    return ret
