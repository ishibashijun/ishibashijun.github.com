descendant_tab

returns descendants of a element matching a tag name

Example


parser = Parser()
parser.open("http://example.com")
ul = parser.tag('ul')
a = ul.descendant_tab('a')
parser.close()

Arguments

  • Name: tag, Type: str
    • tab name

Source


def descendant_tag(self, tag):
    t = tag.lower()
    ret = pyNodeList()
    if self.has_child() == True:
        children = self.children()
        for ch in children:
            if ch.name() == t:
                ret.append(ch)
            descendant = ch.descendant_tag(t)
            if len(descendant) != 0:
                ret.extend(descendant)
    return ret
