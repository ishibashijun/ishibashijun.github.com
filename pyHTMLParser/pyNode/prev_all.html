prev_all

returns elements before this elements that in same DOM hierarchy

Example


parser = Parser()
parser.open("http://example.com")
li = parser.tag('ul').first().child('li').eq(3).prev_all()
parser.close()

Source


def prev_all(self):
    ret = pyNodeList()
    if self.has_parent():
        brothers = self._parent.children()
        for i in range(len(brothers)):
            if brothers[i].name() is not 'comment':
                bro = brothers[i]
                if bro is self: break
                ret.append(bro)
    return ret
