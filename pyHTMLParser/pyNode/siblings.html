siblings

returns siblings of a element

Example


parser = Parser()
parser.open("http://example.com")
li = parser.tag('ul').first().child('li').eq(0).siblings()
parser.close()

Source


def siblings(self):
    ret = pyNodeList()
    if self.has_parent():
        children = self._parent.children()
        for child in children:
            if ch != self:
                ret.append(ch)
    return ret
