def text(self):
    if self._name is not 'comment':
        ret = self._text
        if self.has_child():
            for child in self._children:
                if ret is not '':
                    ret += ' ' + child.text()
                else:
                    ret += child.text()
        return ret
    else:
        return self.comment()
