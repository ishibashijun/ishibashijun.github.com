not_contains

returns elements not containing a specified text

Example


parser = Parser()
parser.open("http://example.com")
ul = parser.tag('ul')
li_not_three = ul.children().not_contains('item3')

Source


def not_contains(self, text):
    ret = pyNodeList()
    for node in self:
        if node.text().find(text) == -1:
            ret.append(node)
    return ret
