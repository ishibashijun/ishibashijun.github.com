not_class

returns elements not matching a class name

Example


parser = Parser()
parser.open("http://example.com")
div = parser.tag('div')
not_links = div.children().not_class('link')
parser.close()

Arguments

  • Name: cls, Type: str
    • class name

Source


def not_class(self, cls):
    ret = pyNodeList()
    cls_name = ('(' + cls + ')')
    for node in self:
        if node.attr('class') is None or cls_name.search(node.attr('class')) is None:
            ret.append(node)
    return ret
