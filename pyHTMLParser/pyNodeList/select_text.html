select_text

returns elements that are type text.

Source


def select_text(self):
    ret = pyNodeList()
    for node in self:
        t = node.attr('type')
        if t is not None and t.lower() == 'text':
            ret.append(node)
    if len(ret) == 1: return ret[0]
    else: return ret
