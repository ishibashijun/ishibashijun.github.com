select_reset

returns elements that are type reset.

Source


def select_reset(self):
    ret = pyNodeList()
    for node in self:
        t = node.attr('type')
        if t is not None and t.lower() == 'reset':
            ret.append(node)
    if len(ret) == 1: return ret[0]
    else: return ret
