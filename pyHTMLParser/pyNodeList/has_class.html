returns true if one or more of the elements has attribute class of the matching value

Example


parser = Parser()
parser.open("http://example.com")
is_main_exists = parser.tag('div').has_class('main')
parser.close()

Arguments

  • Name: c, Type: str
    • class name

Source


def has_class(self, c):
    for node in self:
        n = node.attr('class')
        if n.find(c) != -1:
            return True
    return False
