select_disabled

returns elements that have the attribute disabled and the value disabled or not.

Example


Q_open("http://example.com")
disabled = Q('input').select_disabled()
Q_close()

Source


def select_disabled(self):
    ret = pyNodeList()
    tags = ['button', 'input', 'optgroup', 'option', 'select', 'textarea']
    for node in self:
        if node.name() in tags:
            t = node.attr('disabled')
            if t is not None:
                if t == 'disabled' or t == '':
                    ret.append(node)
    if len(ret) == 1: return ret[0]
    else: return ret
