select_checked

returns input elements that have the attribute checked and the value checked or not.

Example


Q_open("http://example.com")
checked = Q('input').select_checked()
Q_close()

Source


def select_checked(self):
    ret = pyNodeList()
    for node in self:
        t = node.attr('checked')
        if t is not None:
            if t.lower() == 'checked' or t == '':
                ret.append(node)
    if len(ret) == 1: return ret[0]
    else: return ret
