has_cls

returns True if one or more of elements have the attribute class of the matching value.

Example


Q_open("http://example.com")
main_class = Q('div').has_class('main')
Q_close()

Arguments

  • Name: cls, Type: str
    • class name

Source


def has_cls(self, c):
    cls_name = re.compile(c)
    for node in self:
        n = node.attr('class')
        if n is not None and cls_name.search(n) is not None:
            return True
    return False
