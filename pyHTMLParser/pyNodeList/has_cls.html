has_cls

returns true if one or more of the elements has attribute class of the matching value

Example


parser = Parser()
parser.open("http://example.com")
is_main_exists = parser.tag('div').has_class('main')
parser.close()

Arguments

  • Name: cls, Type: str
    • class name

Source


def has_class(self, cls):
    cls_name = (cls)
    for node in self:
        c = node.attr('class')
        if c is not None and cls_name.search(c) is not None:
            return True
    return False
