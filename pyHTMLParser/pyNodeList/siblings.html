siblings

Source


def siblings(self):
    temp = pyNodeList()
    for node in self:
        brothers = node.siblings()
        if len(brothers) is not 0:
            temp.extend(brothers)
    ret = pyNodeList()
    ret.append(temp[0])
    for t in temp:
        for r in ret:
            if t != r:
                ret.append(t)
    return ret
