siblings

returns siblings of each elements

Example


parser = Parser()
parser.open("http://example.com")
li = parser.tag('ul').first().child('li').siblings()
parser.close()

Source


def siblings(self):
    ret = set()
    for node in self:
        brothers = node.siblings()
        if len(brothers) != 0:
            for bro in brothers:
                ret.add(bro)
    if len(ret) != 0: return pyNodeList(ret)
    return pyNodeList()
