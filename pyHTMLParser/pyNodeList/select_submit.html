select_submit

returns elements that are type submit.

Example


Q_open("http://example.com")
submits = Q('input').select_submit()
Q_close()

Source


def select_submit(self):
    ret = pyNodeList()
    for node in self:
        t = node.attr('type')
        if t is not None and t.lower() == 'submit':
            ret.append(node)
    if len(ret) == 1: return ret[0]
    else: return ret
