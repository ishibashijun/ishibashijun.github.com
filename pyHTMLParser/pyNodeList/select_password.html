select_password

returns elements that are type password.

Q_open("http://example.com")

Source


def select_password(self):
    ret = pyNodeList()
    for node in self:
        t = node.attr('type')
        if t is not None and t.lower() == 'password':
            ret.append(node)
    if len(ret) == 1: return ret[0]
    else: return ret
