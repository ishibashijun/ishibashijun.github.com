cls

returns elements matching a class name

Example


parser = Parser()
parser.open("http://example.com")
divs = parser.tag('div')
tags = divs.cls('tag')

Arguments

  • Name: c, Type: str
    • class name

Source


def cls(self, c):
    ret = pyNodeList()
    cls_name = (c)
    for node in self:
        n = node.attr('class')
        if n is not None and cls_name.search(n) is not None:
            ret.append(node)
    return ret
