select_nth_child

returns elements that are the specified index child of their parent.

Example


Q_open("http://example.com")
second_child = Q('div').select_nth_child(2)
Q_close()

Arguments

  • Name: i, Type: int
    • specified index

Source


def select_nth_child(self, i):
    index = i - 1
    parents = list()
    ret = pyNodeList()
    tag = self[0].name()
    for node in self:
        duplicated = False
        if len(parents) != 0:
            for p in parents:
                if p == node.parent():
                    duplicated = True
                    break
        if not duplicated: parents.append(node.parent())
        if tag != node.name():
            return pyNodeList()
    if len(parents) != 0:
        for parent in parents:
            n = parent.child(tag)
            if isinstance(n, pyNodeList) and len(n) != 0: n = n.eq(index)
            if n is not None: ret.append(n)
    if len(ret) == 1: return ret[0]
    else: return ret
