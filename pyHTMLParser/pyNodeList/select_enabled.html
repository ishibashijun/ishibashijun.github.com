select_enabled

returns enabled elements.

Example


Q_open("http://example.com")

Source


def select_enabled(self):
    ret = pyNodeList()
    tags = ['button', 'input', 'optgroup', 'option', 'select', 'textarea']
    for node in self:
        if node.name() in tags:
            t = node.attr('disabled')
            if t is None:
                ret.append(node)
    if len(ret) == 1: return ret[0]
    else: return ret
