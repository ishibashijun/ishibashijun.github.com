tag

returns elements matching a tag name

Example


parser = Parser()
parser.open("http://example.com")
links = parser.cls('link')
a = links.tag('a')

Arguments

  • Name: tag, Type: str

Source


def tag(self, tag):
    t = tag.lower()
    ret = pyNodeList()
    for i in self:
        if i.name() == t:
            ret.append(i)
    return ret
