Q_html

constructs DOM tree from the html.

// DO THINGS

Arguments

  • Name: html, Type: str
    • html str

Source


def Q_html(html):
    global _parser
    if _parser is not None:
        _parser.close()
        _parser.clear()
        _parser.set_html(html)
        _parser.construct()
