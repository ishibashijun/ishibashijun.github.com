Example


Q_open('http://localhost:4000/pyHTMLParser/test.html')

odd = Q('#firstList li:odd')

if isinstance(odd, pyNode):
    print(odd.html())
elif isinstance(odd, pyNodeList):
    for node in odd:
        print(node.html())

Q_close()
